git config

系统级配置 $(prefix)/etc/gitconfig
git config –system user.name junyi

用户级配置 $HOME/.gitconfig
git config –global user.name junyi

项目级配置 .git/config
git config user.name junyi

设置别名 alias
git config –global alias.st status
git config –global alias.br branch
git config –global alias.ci commit
git config –global alias.co checkout

查看已有的配置
git config –system -l
git config –global -l
git config -l

删除一条配置
git config –system –unset user.name

查看帮助
git help config
或
git config
